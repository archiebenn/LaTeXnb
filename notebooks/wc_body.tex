    \section{`wc'}\label{wc}

For printing wordcount, newline count, and byte counts of file

    \subsection{Useful flags}\label{useful-flags}

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{wc} \AttributeTok{{-}l}
\end{Highlighting}
\end{Shaded}

→ Line counts

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{wc} \AttributeTok{{-}w}
\end{Highlighting}
\end{Shaded}

→ Word counts

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{wc} \AttributeTok{{-}c}
\end{Highlighting}
\end{Shaded}

→ Byte counts

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{wc} \AttributeTok{{-}m}
\end{Highlighting}
\end{Shaded}

→ Character counts (UTF-8)

    \subsection{Example uses}\label{example-uses}

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{wc} \AttributeTok{{-}l}\NormalTok{ file.txt}
\end{Highlighting}
\end{Shaded}

→ Outputs line count for file.txt

    \begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{cat}\NormalTok{ file.txt }\KeywordTok{|} \FunctionTok{wc} \AttributeTok{{-}l}
\end{Highlighting}
\end{Shaded}

→ Pipe into word count (same output as above)

    \begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{grep} \StringTok{"apple"}\NormalTok{ file.txt }\KeywordTok{|} \FunctionTok{wc} \AttributeTok{{-}l}
\end{Highlighting}
\end{Shaded}

→ Counts how many lines contain \texttt{apple}
